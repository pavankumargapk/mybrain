IR RECEIVER MODULE
HEXADECIMAL VALUES OF REMOTE BUTTONS
CODE:
#include <IRremote.h>
int RECV_PIN = 8;
IRrecv irrecv(RECV_PIN);
decode_results results;
void setup()
{
  Serial.begin(9600);
  irrecv.enableIRIn(); 
 }
void loop()
 {
  if (irrecv.decode(&results)) 
 {
   Serial.println(results.value, HEX);
   irrecv.resume();
 }
  delay(100);
}
